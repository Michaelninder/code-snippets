# Python example with different syntax elements
import json
from datetime import datetime

class DataProcessor:
    def __init__(self, data_source):
        self.data_source = data_source
        self.processed_data = []
    
    def process_data(self):
        """Process the data from the source"""
        try:
            with open(self.data_source, 'r') as file:
                raw_data = json.load(file)
            
            for item in raw_data:
                processed_item = {
                    'id': item.get('id', 0),
                    'name': item.get('name', '').strip(),
                    'timestamp': datetime.now().isoformat(),
                    'is_valid': bool(item.get('name'))
                }
                self.processed_data.append(processed_item)
                
        except FileNotFoundError:
            print(f"File {self.data_source} not found!")
            return False
        
        return len(self.processed_data) > 0

# Usage example
processor = DataProcessor('data.json')
success = processor.process_data()

if success:
    print(f"Processed {len(processor.processed_data)} items")
